```ts
/** 错误码枚举。与 Rust AppError 完全同步。 */
export enum ErrorCode {
  SVN_EXEC_FAILED = 'SVN_EXEC_FAILED',
  SVN_PARSE_FAILED = 'SVN_PARSE_FAILED',
  INVALID_INPUT = 'INVALID_INPUT',
  IO_ERROR = 'IO_ERROR',
  REPO_ERROR = 'REPO_ERROR',
  SVN_TIMEOUT = 'SVN_TIMEOUT',
  SVN_AUTH_FAILED = 'SVN_AUTH_FAILED',
  SVN_NOT_FOUND = 'SVN_NOT_FOUND',
  SVN_NOT_WORKING_COPY = 'SVN_NOT_WORKING_COPY',
  SVN_OPERATION_IN_PROGRESS = 'SVN_OPERATION_IN_PROGRESS',
  SVN_CANCELLED = 'SVN_CANCELLED',
  NETWORK_UNREACHABLE = 'NETWORK_UNREACHABLE',
  TOOL_NOT_FOUND = 'TOOL_NOT_FOUND',
}

/** 后端错误返回结构 */
export interface ErrorResponse {
  error: string
  message: string
}

/** 错误码 → 中文翻译（i18n 启用前使用） */
export const ERROR_MESSAGES_ZH: Record<ErrorCode, string> = {
  [ErrorCode.SVN_EXEC_FAILED]: 'SVN 命令执行失败',
  [ErrorCode.SVN_PARSE_FAILED]: 'SVN 输出解析错误',
  [ErrorCode.INVALID_INPUT]: '输入参数无效',
  [ErrorCode.IO_ERROR]: '文件系统错误',
  [ErrorCode.REPO_ERROR]: '仓库操作异常',
  [ErrorCode.SVN_TIMEOUT]: '操作超时，请重试',
  [ErrorCode.SVN_AUTH_FAILED]: '认证失败，请检查凭据配置或使用 SSH key / ~/.subversion/auth/ 缓存',
  [ErrorCode.SVN_NOT_FOUND]: 'svn 未找到，请重新安装 Free-SVN',
  [ErrorCode.SVN_NOT_WORKING_COPY]: '不是 SVN 工作副本',
  [ErrorCode.SVN_OPERATION_IN_PROGRESS]: '操作进行中，请稍候',
  [ErrorCode.SVN_CANCELLED]: '操作已取消',
  [ErrorCode.NETWORK_UNREACHABLE]: '网络不可用，请检查网络连接',
  [ErrorCode.TOOL_NOT_FOUND]: '外部工具未找到，请检查设置',
}

/** 从后端错误响应提取消息 */
export function getErrorMessage(err: unknown): string {
  if (err && typeof err === 'object') {
    const e = err as Record<string, unknown>
    if (typeof e.error === 'string') {
      return ERROR_MESSAGES_ZH[e.error as ErrorCode] || (e.message as string) || 'Unknown error'
    }
  }
  return typeof err === 'string' ? err : 'Unknown error'
}
```
