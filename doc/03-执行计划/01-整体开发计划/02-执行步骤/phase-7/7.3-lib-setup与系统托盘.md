### 7.3A — 修改 `src-tauri/src/lib.rs`

完整 setup 函数，包含插件注册、系统托盘、单实例防护：

```rust
use tauri::{
    AppHandle, Manager, WebviewUrl, WebviewWindowBuilder,
    menu::{MenuBuilder, MenuItemBuilder, SubmenuBuilder},
    tray::{TrayIconBuilder, TrayIconEvent, MouseButton, MouseButtonState},
    Emitter,
};

#[cfg_attr(mobile, tauri::mobile_entry_point)]
pub fn run() {
    // 插件注册（setup 前必须完成的 5 个插件）
    let builder = tauri::Builder::default()
        .plugin(tauri_plugin_store::Builder::new().build())
        .plugin(tauri_plugin_single_instance::init(|app, _args, _cwd| {
            // 第二实例启动时激活首实例窗口
            if let Some(window) = app.get_webview_window("main") {
                let _ = window.show();
                let _ = window.set_focus();
            }
        }))
        .plugin(tauri_plugin_window_state::Builder::default().build())
        .plugin(
            tauri_plugin_log::Builder::new()
                .target(tauri_plugin_log::Target::new(
                    tauri_plugin_log::TargetKind::LogDir { name: "logs" },
                ))
                .level(log::LevelFilter::Info)
                .build(),
        )
        .plugin(tauri_plugin_dialog::init())
        .manage(crate::svn::queue::SvnQueue::new())
        .setup(|app| {
            // ── 1. 初始化日志系统 ──
            crate::logging::logger::init(app.handle())?;

            // ── 2. 初始化配置存储 ──
            crate::config::store::init(app.handle())?;

            // ── 3. 构建系统托盘菜单 ──
            let show = MenuItemBuilder::with_id("show", "Show Main Window")
                .accelerator("CmdOrCtrl+Shift+S").build(app)?;
            let update = MenuItemBuilder::with_id("update", "SVN Update").build(app)?;
            let cleanup = MenuItemBuilder::with_id("cleanup", "SVN Cleanup").build(app)?;
            let about = MenuItemBuilder::with_id("about", "About Free-SVN").build(app)?;
            let quit = MenuItemBuilder::with_id("quit", "Quit").accelerator("CmdOrCtrl+Q").build(app)?;

            let menu = MenuBuilder::new(app)
                .item(&show)
                .separator()
                .item(&update)
                .item(&cleanup)
                .separator()
                .item(&about)
                .separator()
                .item(&quit)
                .build()?;

            // ── 4. 创建系统托盘图标 ──
            TrayIconBuilder::new()
                .icon(app.default_window_icon().unwrap().clone())
                .menu(&menu)
                .tooltip("Free-SVN")
                .on_menu_event(move |app, event| {
                    match event.id().as_ref() {
                        "show" => {
                            if let Some(window) = app.get_webview_window("main") {
                                let _ = window.show();
                                let _ = window.set_focus();
                            }
                        }
                        "update" => {
                            let cwd = crate::config::store::current_workspace()
                                .unwrap_or_default();
                            if !cwd.is_empty() {
                                tauri::async_runtime::spawn(async move {
                                    let _ = crate::commands::update_workspace(
                                        app.clone(),
                                        crate::commands::UpdateParams {
                                            path: cwd, revision: None,
                                            depth: None, ignore_externals: None,
                                            credentials: None,
                                        },
                                        app.state::<crate::svn::queue::SvnQueue>(),
                                    ).await;
                                });
                            }
                        }
                        "cleanup" => {
                            let cwd = crate::config::store::current_workspace()
                                .unwrap_or_default();
                            if !cwd.is_empty() {
                                tauri::async_runtime::spawn(async move {
                                    let _ = crate::commands::cleanup_workspace(
                                        app.clone(),
                                        crate::commands::ExportParams {
                                            path: cwd, target_dir: String::new(),
                                            revision: None, ignore_externals: None,
                                            credentials: None,
                                        },
                                    ).await;
                                });
                            }
                        }
                        "about" => {
                            if let Some(window) = app.get_webview_window("main") {
                                let _ = window.show();
                                let _ = window.set_focus();
                                let _ = window.emit("navigate", "/about");
                            }
                        }
                        "quit" => { app.exit(0); }
                        _ => {}
                    }
                })
                .on_tray_icon_event(|tray, event| {
                    // 单击托盘图标 → 激活/显示主窗口
                    if let TrayIconEvent::Click {
                        button: MouseButton::Left,
                        button_state: MouseButtonState::Up, ..
                    } = event {
                        if let Some(app) = tray.app_handle() {
                            if let Some(window) = app.get_webview_window("main") {
                                let _ = window.show();
                                let _ = window.set_focus();
                            }
                        }
                    }
                })
                .build(app)?;

            // ── 5. 注册 Shell 右键菜单 ──
            let _ = crate::shell_integration::register();

            Ok(())
        })
        .invoke_handler(tauri::generate_handler![
            // 阶段三注册的 26 个 commands
            crate::commands::get_status, crate::commands::get_info,
            crate::commands::get_diff,
            crate::commands::get_log, crate::commands::get_blame,
            crate::commands::checkout_repo, crate::commands::update_workspace,
            crate::commands::create_commit,
            crate::commands::add_files, crate::commands::delete_files,
            crate::commands::revert_files, crate::commands::resolve_conflict,
            crate::commands::set_ignore,
            crate::commands::switch_branch, crate::commands::copy_branch_tag,
            crate::commands::merge_branch,
            crate::commands::cleanup_workspace, crate::commands::export_workspace,
            crate::commands::lock_files, crate::commands::unlock_files,
            crate::commands::relocate_repo, crate::commands::property_ops,
            crate::commands::cancel_operation, crate::commands::check_network,
            crate::commands::get_logs, crate::commands::export_logs,
            crate::commands::load_settings, crate::commands::save_settings,
            crate::commands::test_connection, crate::commands::save_credentials,
            crate::commands::clear_credentials,
        ]);

    builder.run(tauri::generate_context!())
        .expect("error while running tauri application");
}
```

**7.3B — 关闭窗口→最小化到托盘（App.vue 前端侧）：**

```ts
import { getCurrentWebviewWindow } from '@tauri-apps/api/webviewWindow'

onMounted(() => {
  const win = getCurrentWebviewWindow()
  // 关闭按钮隐藏窗口到托盘而非退出
  win.onCloseRequested(async () => {
    await win.hide()
  })
})
```
