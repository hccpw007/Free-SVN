### 7.3A — 修改 `src-tauri/src/lib.rs`

完整 setup 函数，包含插件注册、系统托盘、单实例防护：

```rust
use tauri::{
    AppHandle, Manager, WebviewUrl, WebviewWindowBuilder,
    menu::{MenuBuilder, MenuItemBuilder, SubmenuBuilder},
    tray::{TrayIconBuilder, TrayIconEvent, MouseButton, MouseButtonState},
    Emitter,
};

#[cfg_attr(mobile, tauri::mobile_entry_point)]
pub fn run() {
    // 插件注册（setup 前必须完成的 5 个插件）
    let builder = tauri::Builder::default()
        .plugin(tauri_plugin_store::Builder::new().build())
        .plugin(tauri_plugin_single_instance::init(|app, args, _cwd| {
            // 第二实例启动时激活首实例窗口
            if let Some(window) = app.get_webview_window("main") {
                let _ = window.show();
                let _ = window.set_focus();

                // 解析 --svn-cmd 参数（右键菜单触发）：格式为 --svn-cmd <command> [file1 file2 ...]
                // 将命令和文件路径通过事件传递给主窗口，由前端 svnStore 处理执行
                if let Some(cmd_idx) = args.iter().position(|a| a == "--svn-cmd") {
                    let cmd = args.get(cmd_idx + 1).cloned().unwrap_or_default();
                    let files: Vec<String> = args.iter()
                        .skip(cmd_idx + 2)
                        .filter(|a| !a.starts_with('--'))
                        .cloned()
                        .collect();
                    if !cmd.is_empty() {
                        let _ = window.emit("svn:shell-cmd", serde_json::json!({
                            "command": cmd,
                            "files": files,
                        }));
                    }
                }
            }
        }))
        .plugin(tauri_plugin_window_state::Builder::default().build())
        .plugin(
            tauri_plugin_log::Builder::new()
                .target(tauri_plugin_log::Target::new(
                    tauri_plugin_log::TargetKind::LogDir { name: "logs" },
                ))
                .level(log::LevelFilter::Info)
                .build(),
        )
        .plugin(tauri_plugin_dialog::init())
        .plugin(tauri_plugin_shell::init())
        .plugin(tauri_plugin_clipboard_manager::init())
        .manage(crate::svn::queue::SvnQueue::new())
        .setup(|app| {
            // ── 1. 初始化日志系统 ──
            crate::logging::logger::init(app.handle())?;

            // ── 2. 初始化配置存储 ──
            crate::config::store::init(app.handle())?;

            // ── 3. 构建系统托盘菜单 ──
            let show = MenuItemBuilder::with_id("show", "Show Main Window")
                .accelerator("CmdOrCtrl+Shift+S").build(app)?;
            let update = MenuItemBuilder::with_id("update", "SVN Update").build(app)?;
            let cleanup = MenuItemBuilder::with_id("cleanup", "SVN Cleanup").build(app)?;
            let about = MenuItemBuilder::with_id("about", "About Free-SVN").build(app)?;
            let quit = MenuItemBuilder::with_id("quit", "Quit").accelerator("CmdOrCtrl+Q").build(app)?;

            let menu = MenuBuilder::new(app)
                .item(&show)
                .separator()
                .item(&update)
                .item(&cleanup)
                .separator()
                .item(&about)
                .separator()
                .item(&quit)
                .build()?;

            // ── 4. 创建系统托盘图标 ──
            TrayIconBuilder::new()
                .icon(app.default_window_icon().unwrap().clone())
                .menu(&menu)
                .tooltip("Free-SVN")
                .on_menu_event(move |app, event| {
                    match event.id().as_ref() {
                        "show" => {
                            if let Some(window) = app.get_webview_window("main") {
                                let _ = window.show();
                                let _ = window.set_focus();
                            }
                        }
                        "update" => {
                            let cwd = crate::config::store::current_workspace()
                                .unwrap_or_default();
                            if !cwd.is_empty() {
                                tauri::async_runtime::spawn(async move {
                                    let _ = crate::commands::update_workspace(
                                        app.clone(),
                                        crate::commands::UpdateParams {
                                            path: cwd, revision: None,
                                            depth: None, ignore_externals: None,
                                            credentials: None,
                                        },
                                        app.state::<crate::svn::queue::SvnQueue>(),
                                    ).await;
                                });
                            }
                        }
                        "cleanup" => {
                            let cwd = crate::config::store::current_workspace()
                                .unwrap_or_default();
                            if !cwd.is_empty() {
                                tauri::async_runtime::spawn(async move {
                                    let _ = crate::commands::cleanup_workspace(
                                        cwd,
                                        app.state::<crate::svn::queue::SvnQueue>(),
                                    ).await;
                                });
                            }
                        }
                        "about" => {
                            if let Some(window) = app.get_webview_window("main") {
                                let _ = window.show();
                                let _ = window.set_focus();
                                let _ = window.emit("navigate", "/about");
                            }
                        }
                        "quit" => { app.exit(0); }
                        _ => {}
                    }
                })
                .on_tray_icon_event(|tray, event| {
                    // 单击托盘图标 → 激活/显示主窗口
                    if let TrayIconEvent::Click {
                        button: MouseButton::Left,
                        button_state: MouseButtonState::Up, ..
                    } = event {
                        if let Some(app) = tray.app_handle() {
                            if let Some(window) = app.get_webview_window("main") {
                                let _ = window.show();
                                let _ = window.set_focus();
                            }
                        }
                    }
                })
                .build(app)?;

            // ── 5. 注册 Shell 右键菜单 ──
            let _ = crate::shell_integration::register();

            Ok(())
        })
        .invoke_handler(tauri::generate_handler![
            // 阶段三注册的 26 个 commands
            crate::commands::get_status, crate::commands::get_info,
            crate::commands::get_diff,
            crate::commands::get_log, crate::commands::get_blame,
            crate::commands::checkout_repo, crate::commands::update_workspace,
            crate::commands::create_commit,
            crate::commands::add_files, crate::commands::delete_files,
            crate::commands::revert_files, crate::commands::resolve_conflict,
            crate::commands::set_ignore,
            crate::commands::switch_branch, crate::commands::copy_branch_tag,
            crate::commands::merge_branch,
            crate::commands::cleanup_workspace, crate::commands::export_workspace,
            crate::commands::lock_files, crate::commands::unlock_files,
            crate::commands::relocate_repo, crate::commands::property_ops,
            crate::commands::cancel_operation, crate::commands::check_network,
            crate::commands::get_logs, crate::commands::export_logs,
            crate::commands::load_settings, crate::commands::save_settings,
            crate::commands::test_connection, crate::commands::save_credentials,
            crate::commands::clear_credentials,
        ]);

    builder.run(tauri::generate_context!())
        .expect("error while running tauri application");
}
```

**Linux 平台系统托盘依赖说明：**

系统托盘在 Linux 上需要以下系统库的支持：

- Debian/Ubuntu：`apt install libayatana-appindicator3-dev`
- Fedora：`dnf install libappindicator-gtk3-devel`

Tauri 2 通过 `tauri-plugin-tray` 或原生 `tray-icon` 功能自动链接这些库。如果上述依赖未安装，cargo build 会在链接阶段失败。确认依赖后方可执行 `cnpm run tauri build`。

**7.3C — 系统托盘操作中徽章（业务设计 §3.4 要求）：**

操作进行中时，在系统托盘图标上显示小徽章（macOS 支持设置 badge 文本，Windows/Linux 可通过动态图标替换实现）：

- **macOS**：使用 `app.tray_icon_by_id("main")?.set_icon_as_template(true)` + `tray.set_title("●")` 或 `app.get_tray_icon_by_id("main")?.set_tooltip("SVN 操作进行中...")`。徽章文字通过 `tray.set_icon()` 切换为带红点的图标或使用 `NSApp.dockTile.badgeLabel`（通过 tauri-plugin-shell 调用 AppleScript）
- **Windows/Linux**：切换托盘图标为带"进度"或"警告"标记的版本。SVN 长时间操作（checkout/update/merge）开始时调用 `tray.set_icon()` 切换到带徽章的图标，操作完成后恢复原始图标
- **实现时机**：在 `svnStore.subscribeToOperations()` 中监听 `operation:started` / `operation:completed` / `operation:error` 三个事件，通过 invoke 调用 Rust 端的 `set_tray_badge(visible: bool)` 命令切换图标状态（该命令在 `lib.rs` 或独立的 `tray.rs` 中注册为 tauri command）

> **当前阶段简化处理：** 由于托盘图标动态切换需要额外图标资源，第一阶段仅在 `cleanup`/`update` 等长操作执行前修改托盘 tooltip 为 "SVN 操作进行中..."，操作完成后恢复 "Free-SVN"。图标徽章增强可留待后续迭代实现。

**7.3D — 后台操作完成系统通知（P1-4 / 业务设计 §3.4）：**

当窗口隐藏到托盘（后台模式）时，SVN 操作完成后通过操作系统通知中心发出通知，确保用户感知操作结果：

- **macOS**：使用 `tauri-plugin-notification`（或 `tauri_plugin_shell` 调用 `osascript -e 'display notification "..." with title "Free-SVN"'`）
- **Windows**：使用 `tauri-plugin-notification` 调用 Windows Toast 通知（通过 Windows Action Center）
- **Linux**：使用 `tauri-plugin-notification` 调用 `notify-send` 命令

```rust
// 在 lib.rs 或独立的 notification.rs 中 — 通知发送辅助函数
// 当前使用 tauri-plugin-shell 调用系统命令发送通知
fn send_os_notification(app: &AppHandle, title: &str, body: &str) {
    let script = format!(
        "display notification \"{}\" with title \"Free-SVN\" subtitle \"{}\"",
        body.replace('"', "\\\""),
        title.replace('"', "\\\"")
    );
    // macOS: osascript / Windows: powershell toast / Linux: notify-send
    #[cfg(target_os = "macos")]
    let _ = std::process::Command::new("osascript")
        .args(["-e", &script])
        .spawn();
    #[cfg(target_os = "linux")]
    let _ = std::process::Command::new("notify-send")
        .args(["Free-SVN", &format!("{}: {}", title, body)])
        .spawn();
}
```

在长操作完成的回调中触发：
1. 后端长操作完成时调用 `send_os_notification()`
2. 或者前端 svnStore 监听 `operation:completed`/`operation:error` 事件，在当前窗口被隐藏时通过 `@tauri-apps/plugin-shell` 调用系统命令

**注意：** 当前阶段不引入 `tauri-plugin-notification` 依赖，使用 `tauri-plugin-shell` 的 shell 命令方式发送通知（macOS `osascript` / Windows `powershell` / Linux `notify-send`）。完整 `tauri-plugin-notification` 能力可在后续迭代中切换。
