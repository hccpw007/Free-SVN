## 操作细节

### 7.4A — mod.rs

```rust
pub mod macos; pub mod windows; pub mod linux;
pub fn register() -> Result<(), AppError> {
    #[cfg(target_os = "macos")] macos::register()?;
    #[cfg(target_os = "windows")] windows::register()?;
    #[cfg(target_os = "linux")] linux::register()?;
    Ok(())
}
pub fn unregister() -> Result<(), AppError> {
    #[cfg(target_os = "macos")] macos::unregister()?;
    #[cfg(target_os = "windows")] windows::unregister()?;
    #[cfg(target_os = "linux")] linux::unregister()?;
    Ok(())
}
```
