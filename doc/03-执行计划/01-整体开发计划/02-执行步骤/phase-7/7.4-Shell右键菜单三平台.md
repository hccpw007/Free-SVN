### 7.4B — macos.rs

通过 Automator `.workflow` bundle 注册 Finder 右键服务：

```rust
use std::path::PathBuf;
use std::fs;
#[cfg(unix)] use std::os::unix::fs::PermissionsExt;

/// 注册工作副本目录 12 项菜单 + 普通文件夹 1 项菜单
pub fn register() -> Result<(), AppError> {
    let home = dirs::home_dir().ok_or(AppError::Repo("无法获取用户目录".into())))?;
    let svc_dir = home.join("Library/Services");

    // 生成 12 项 SVN 操作的工作流配置文件（名称, 命令）
    // macOS Finder 服务菜单项名称（使用英文以保持跨区域兼容性；
    // 如需本地化，用户可在 Automator 中手动修改 .workflow 的 NSMenuItem）
    const MENUS: &[(&str, &str)] = &[
        ("Commit", "commit"), ("Update", "update"), ("Log", "log"),
        ("Update to Revision", "update-rev"), ("Diff", "diff"), ("Revert", "revert"),
        ("Switch", "switch"), ("Branch/Tag", "branch-tag"), ("Merge", "merge"),
        ("Cleanup", "cleanup"), ("Export", "export"),
    ];
    for (name, cmd) in MENUS {
        let plist = format!(r#"<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN"
  "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
    <key>NSServices</key>
    <array>
        <dict>
            <key>NSMenuItem</key>
            <dict><key>default</key><string>SVN {name}</string></dict>
            <key>NSMessage</key><string>runWorkflowAsService</string>
            <key>NSRequiredFiles</key><integer>1</integer>
        </dict>
    </array>
</dict>
</plist>"#);
        let bundle = svc_dir.join(format!("SVN-{name}.workflow"));
        if !bundle.exists() {
            fs::create_dir_all(bundle.join("Contents"))?;
            fs::write(bundle.join("Contents/Info.plist"), &plist)?;
            let wflow = format!(
                r#"<?xml version="1.0" encoding="UTF-8"?>
<plist version="1.0"><dict>
<key>AMWorkflowActions</key><array><dict>
<key>action</key><key>AMAction</key><dict>
<key>AMActionName</key><string>Run Shell Script</string>
<key>AMAccepts</key><dict><key>Container</key><string>List</string>
<key>Optional</key><false/>
<key>Types</key><array><string>public.file-url</string></array></dict>
<key>AMParameterProperties</key><dict>
<key>COMMAND_STRING</key><string>/Applications/Free-SVN.app/Contents/MacOS/free-svn --svn-cmd {cmd} "$@"</string>
<key>inputMethod</key><integer>1</integer>
<key>shell</key><integer>0</integer>
</dict></dict></dict></array>
<key>AMWorkflowMetadata</key><dict>
<key>AMWorkflowSummary</key><string>SVN {name}</string>
</dict></dict></plist>"#,
                cmd = cmd, name = name,
            );
            fs::write(bundle.join("Contents/document.wflow"), &wflow)?;
        }
    }
    // 普通文件夹 1 项（检索）
    // 检出操作使用英文名（跨区域兼容）
    let name_checkout = "Checkout";
    let bundle = svc_dir.join(format!("SVN-{}.workflow", name_checkout));
    if !bundle.exists() {
        let plist_checkout = format!(r#"<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN"
  "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
    <key>NSServices</key>
    <array>
        <dict>
            <key>NSMenuItem</key><dict><key>default</key><string>SVN Checkout</string></dict>
            <key>NSMessage</key><string>runWorkflowAsService</string>
            <key>NSRequiredFiles</key><integer>1</integer>
        </dict>
    </array>
</dict>
</plist>"#);
        fs::create_dir_all(bundle.join("Contents"))?;
        fs::write(bundle.join("Contents/Info.plist"), &plist_checkout)?;
        let wflow_checkout = format!(
            r#"<?xml version="1.0" encoding="UTF-8"?>
<plist version="1.0"><dict>
<key>AMWorkflowActions</key><array><dict>
<key>action</key><key>AMAction</key><dict>
<key>AMActionName</key><string>Run Shell Script</string>
<key>AMAccepts</key><dict><key>Container</key><string>List</string>
<key>Optional</key><false/>
<key>Types</key><array><string>public.file-url</string></array></dict>
<key>AMParameterProperties</key><dict>
<key>COMMAND_STRING</key><string>/Applications/Free-SVN.app/Contents/MacOS/free-svn --svn-cmd checkout "$@"</string>
<key>inputMethod</key><integer>1</integer>
<key>shell</key><integer>0</integer>
</dict></dict></dict></array>
<key>AMWorkflowMetadata</key><dict>
<key>AMWorkflowSummary</key><string>SVN Checkout</string>
</dict></dict></plist>"#);
        fs::write(bundle.join("Contents/document.wflow"), &wflow_checkout)?;
    }

    // 通知 Finder 刷新服务
    std::process::Command::new("/System/Library/CoreServices/pbs").arg("-flush").output()?;
    Ok(())
}

pub fn unregister() -> Result<(), AppError> {
    let svc_dir = dirs::home_dir().ok_or(AppError::Repo("无法获取用户目录".into())))?.join("Library/Services");
    for entry in fs::read_dir(&svc_dir)? {
        let entry = entry?;
        if entry.file_name().to_string_lossy().starts_with("SVN-") && entry.path().extension() == Some("workflow") {
            fs::remove_dir_all(entry.path())?;
        }
    }
    // 刷新服务缓存
    std::process::Command::new("/System/Library/CoreServices/pbs").arg("-flush").output()?;
    Ok(())
}
```

### 7.4C — windows.rs

通过注册表 `HKEY_CLASSES_ROOT\Directory\Background\shell\`、`HKEY_CLASSES_ROOT\*\shell\` 和 `HKEY_CLASSES_ROOT\Directory\shell\` 注册：

```rust
use std::process::Command;

const APP: &str = "Free-SVN";

/// 注册工作副本右键菜单（12 项）
pub fn register() -> Result<(), AppError> {
    // 注册表基路径
    const WC_KEY: &str = r"HKEY_CLASSES_ROOT\Directory\Background\shell";
    const DIR_KEY: &str = r"HKEY_CLASSES_ROOT\Directory\shell";
    const FILE_KEY: &str = r"HKEY_CLASSES_ROOT\*\shell";

    // 辅助：添加菜单项到注册表
    fn add_menu(parent: &str, name: &str, cmd: &str, icon: &str) -> Result<(), AppError> {
        let key = format!(r"{}\SVN-{}", parent, name);
        Command::new("reg")
            .args(&["add", &key, "/ve", "/t", "REG_SZ", "/d", &format!("SVN {}", name), "/f"])
            .output()?;
        Command::new("reg")
            .args(&["add", &format!(r"{}\command", key), "/ve", "/t", "REG_SZ", "/d",
                &format!(r#""{}" --svn-cmd {} "%V""#, std::env::current_exe()?.display(), cmd), "/f"])
            .output()?;
        if !icon.is_empty() {
            Command::new("reg")
                .args(&["add", &key, "/v", "Icon", "/t", "REG_SZ", "/d", icon, "/f"])
                .output()?;
        }
        Ok(())
    }

    // 工作副本 12 项——同时在目录背景(WC_KEY)和文件(FILE_KEY)下注册
    let menus = [
        ("Commit", "commit"), ("Update", "update"), ("Log", "log"),
        ("Update to Rev", "update-rev"), ("Diff", "diff"), ("Revert", "revert"),
        ("Switch", "switch"), ("Branch/Tag", "branch-tag"), ("Merge", "merge"),
        ("Cleanup", "cleanup"), ("Export", "export"),
    ];
    for (name, cmd) in &menus {
        add_menu(WC_KEY, name, cmd, "")?;
        add_menu(FILE_KEY, name, cmd, "")?;  // 文件级右键菜单
    }
    // 普通文件夹 1 项（Directory\shell 下，右键文件夹本身）
    add_menu(DIR_KEY, "检出", "checkout", "")?;

    Ok(())
}

pub fn unregister() -> Result<(), AppError> {
    for parent in &[
        r"HKEY_CLASSES_ROOT\Directory\Background\shell",
        r"HKEY_CLASSES_ROOT\Directory\shell",
        r"HKEY_CLASSES_ROOT\*\shell",
    ] {
        let output = Command::new("reg")
            .args(&["query", parent, "/f", "SVN-*", "/k"]).output()?;
        for line in String::from_utf8_lossy(&output.stdout).lines() {
            let line = line.trim();
            if line.contains("SVN-") {
                Command::new("reg").args(&["delete", line, "/f"]).output()?;
            }
        }
    }
    Ok(())
}
```

> **注意：** Windows 注册表写入需管理员权限。构建时应配置 `requested-execution-level="requireAdministrator"`，或运行首次 setup 时提权。
>
> **区域兼容性：** `reg query` 输出格式随 Windows 区域设置变化。当前匹配逻辑 `line.contains("SVN-")` 理论上兼容所有区域，因为用户菜单名使用英文 `"SVN-{name}"` 格式不依赖区域文字。但 `reg delete` 的行提取仍需验证。稳健方案如下（推荐 v2 迁移计划）：
>
> **推荐替代方案**（使用 `rust-winreg` crate，避免 `reg.exe` 区域兼容性风险）：
> ```rust
> // 使用 winreg crate 替代 reg.exe 命令行
> use winreg::enums::*;

>

>     // 使用 HKCU\Software\Classes 而非 HKCR（无需管理员权限）
>     let classes = hkcu.open_subkey_with_flags("Software\Classes", KEY_WRITE)?;

>         let key_path = format!("Directory\Background\shell\SVN-{}", name);

>         let (cmd_key, _) = classes.create_subkey(format!("{}\command", key_path))?;

### 7.4D — linux.rs

通过 Nautilus 脚本注册：

```rust
use std::fs;
#[cfg(unix)] use std::os::unix::fs::PermissionsExt;

/// 注册工作副本右键菜单（12 项）
pub fn register() -> Result<(), AppError> {
    let script_dir = dirs::home_dir()
        .ok_or(AppError::Repo("无法获取用户目录".into())))?
        .join(".local/share/nautilus/scripts/SVN");
    fs::create_dir_all(&script_dir)?;

    let exe = std::env::current_exe().map_err(|_| AppError::Repo("无法获取可执行路径".into()))?;

    // 12 项 SVN 操作脚本
    let menus = [
        ("Commit", "commit"), ("Update", "update"), ("Log", "log"),
        ("Update to Rev", "update-rev"), ("Diff", "diff"), ("Revert", "revert"),
        ("Switch", "switch"), ("Branch/Tag", "branch-tag"), ("Merge", "merge"),
        ("Cleanup", "cleanup"), ("Export", "export"),
    ];
    for (label, cmd) in &menus {
        let script = script_dir.join(label);
        let content = format!(
            "#!/bin/bash\nFILES=()\nwhile IFS= read -r file; do FILES+=(\"$file\"); done <<< \"$NAUTILUS_SCRIPT_SELECTED_FILE_PATHS\"\n\"{}\" --svn-cmd {} \"${{FILES[@]}}\"\n",
            exe.display(), cmd
        );
        fs::write(&script, &content)?;
        fs::set_permissions(&script, std::fs::Permissions::from_mode(0o755))?;
    }

    // 普通文件夹 1 项（检出）
    let checkout = script_dir.join("Checkout");
    fs::write(&checkout, format!(
        "#!/bin/bash\nFILES=()\nwhile IFS= read -r file; do FILES+=(\"$file\"); done <<< \"$NAUTILUS_SCRIPT_SELECTED_FILE_PATHS\"\n\"{}\" --svn-cmd checkout \"${{FILES[@]}}\"\n",
        exe.display()
    ))?;
    fs::set_permissions(&checkout, std::fs::Permissions::from_mode(0o755))?;

    Ok(())
}

pub fn unregister() -> Result<(), AppError> {
    let script_dir = dirs::home_dir()
        .ok_or(AppError::Repo("无法获取用户目录".into())))?
        .join(".local/share/nautilus/scripts/SVN");
    if script_dir.exists() {
        fs::remove_dir_all(&script_dir)?;
    }
    Ok(())
}
```

> **注意：** Nautilus 脚本目录 `~/.local/share/nautilus/scripts/` 在 GNOME 文件管理器下生效。其他文件管理器（Dolphin、Thunar）路径不同，可通过 `XDG_CONFIG_HOME` 环境变量检测。
