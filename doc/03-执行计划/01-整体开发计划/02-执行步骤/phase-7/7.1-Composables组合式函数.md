### 7.1A — useKeyboardShortcuts.ts

全局快捷键注册，3 级作用域（dialog>panel>global），ProgressOverlay 期间仅 Esc。

```ts
import { onMounted, onUnmounted } from 'vue'
import { useRouter } from 'vue-router'
import { useFileListStore } from '@/stores/fileList'

type ShortcutHandler = (e: KeyboardEvent) => void
interface Shortcut { id?: string; keys: string; handler: ShortcutHandler; scope: 'dialog' | 'panel' | 'global' }

const allShortcuts: Shortcut[] = []

export function useKeyboardShortcuts() {
  const router = useRouter()
  const fileListStore = useFileListStore()

  function register(s: Shortcut[]) { allShortcuts.push(...s) }

  function unregister(id: string) {
    for (let i = allShortcuts.length - 1; i >= 0; i--) {
      if (allShortcuts[i].id === id) allShortcuts.splice(i, 1)
    }
  }

  function handler(e: KeyboardEvent) {
    if (fileListStore.isOperationRunning && e.key !== 'Escape') return
    const key = e.key.toLowerCase()
    const ctrl = e.ctrlKey || e.metaKey
    for (const s of allShortcuts) {
      if (s.keys === key || (ctrl && s.keys === `ctrl+${key}`)) {
        e.preventDefault(); s.handler(e); return
      }
    }
  }

  onMounted(() => window.addEventListener('keydown', handler))
  onUnmounted(() => window.removeEventListener('keydown', handler))
  return { register, unregister }
}
```

**默认快捷键（注册时机在 App.vue 或各页面 onMounted）：**

| 快捷键 | 操作 | 作用域 | 注册时机 | 宿主组件 |
|--------|------|--------|---------|---------|
| Ctrl+Enter | 提交 | panel | onMounted → register | CommitPage.vue |
| Ctrl+D | 打开差异 | global | onMounted → register | App.vue |
| Ctrl+R | 刷新文件列表 | global | onMounted → register | App.vue |
| Ctrl+, | 打开设置 | global | onMounted → register | App.vue |
| Ctrl+W | 关闭面板 | global | onMounted → register | App.vue |
| Esc | 关闭对话框/面板 | dialog | onMounted → register | App.vue |
| Ctrl+F | diff 搜索 | panel | onMounted → register，onUnmounted → unregister | DiffViewer.vue |
| Ctrl+Down/Up | 导航差异块 | panel | onMounted → register，onUnmounted → unregister | DiffViewer.vue |

> **注册规则：** 面板级快捷键（Ctrl+F / Ctrl+Down/Up / Ctrl+Enter）在宿主组件挂载时 register，卸载时 unregister。全局快捷键在 App.vue onMounted 一次性注册。所有 register 调用时传入带 `id` 的 Shortcut 对象（如 `{ id: 'diff-search', keys: 'ctrl+f', handler: ..., scope: 'panel' }`），便于组件销毁时定向 unregister(id)。

### 7.1B — useFileSelection.ts

Shift+Click 范围选择 + Ctrl/Cmd+Click 切换。

```ts
import { ref } from 'vue'

export function useFileSelection() {
  const lastIdx = ref(-1)

  function handleClick(idx: number, e: MouseEvent, items: {path:string}[], selected: Set<string>, onChg: (s:Set<string>)=>void) {
    const s = new Set(selected)
    if (e.shiftKey && lastIdx.value >= 0) {
      for (let i = Math.min(lastIdx.value, idx); i <= Math.max(lastIdx.value, idx); i++) s.add(items[i].path)
    } else if (e.ctrlKey || e.metaKey) {
      s.has(items[idx].path) ? s.delete(items[idx].path) : s.add(items[idx].path)
    } else { s.clear(); s.add(items[idx].path) }
    lastIdx.value = idx; onChg(s)
  }

  return { handleClick, reset: () => { lastIdx.value = -1 } }
}
```

### 7.1C — useAutoRefresh.ts

window focus → 300ms 防抖 → fileListStore.refresh()。

```ts
import { onMounted, onUnmounted } from 'vue'
import { useFileListStore } from '@/stores/fileList'
export function useAutoRefresh() {
  const fl = useFileListStore()
  let t: ReturnType<typeof setTimeout> | null = null
  const f = () => { if (t) clearTimeout(t); t = setTimeout(() => fl.refresh(), 300) }
  onMounted(() => window.addEventListener('focus', f))
  onUnmounted(() => { window.removeEventListener('focus', f); if (t) clearTimeout(t) })
}
```

### 7.1D — useOperationGuard.ts

读取 `fileListStore.isOperationRunning` 提供全局禁用状态。

```ts
import { computed } from 'vue'
import { useFileListStore } from '@/stores/fileList'
export function useOperationGuard() {
  const fileListStore = useFileListStore()
  const blocked = computed(() => fileListStore.isOperationRunning)
  return { isBlocked: blocked, tooltip: computed(() => blocked.value ? '操作进行中，请等待' : '') }
}
```

> **ToolBar 协作方式：** ToolBar.vue 使用 `isBlocked` 控制按钮 `disabled` 属性；禁用态样式 `opacity-50 cursor-not-allowed`。溢出折叠通过 `ResizeObserver` 监测容器宽度，动态计算可见按钮数量，溢出按钮折叠到 `···` 更多菜单（El-dropdown）。至少保留"提交"和"设置"按钮始终可见。

### 7.1E — useNetworkStatus.ts

检测网络可达性，更新 `workspaceStore.isOffline`。

```ts
import { useWorkspaceStore } from '@/stores/workspace'
import { checkNetwork } from '@/services/svn'

export function useNetworkStatus() {
  const ws = useWorkspaceStore()
  async function check(): Promise<boolean> {
    const ok = await checkNetwork()
    ws.isOffline = !ok
    return ok
  }
  return { checkNetwork: check }
}
```

> **注意：** 网络检测通过 `services/svn.ts` 的 `checkNetwork` 方法调用后端 `check_network` 命令，不直接 `import { invoke }`，确保前端唯一 invoke 入口规则。
