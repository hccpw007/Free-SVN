## 操作细节

### 5.3A — 新建 `src/components/common/ToolBar.vue`

高度 40px（h-10），背景 `bg-white dark:bg-slate-800`，底部 1px 边框。使用 `<script setup lang="ts">` 语法，Lucide 图标（无 emoji）。

**完整按钮列表（10 个）：**

| 标签 | Lucide 图标 | 操作 | 禁用条件 | 禁用 tooltip |
|------|------------|------|---------|-------------|
| 提交 | GitCommit | router.push('/workspace/commit') | 无工作副本 / 无变更 | "无可用工作副本" / "当前工作副本无变更" |
| 更新 | RefreshCw | svnStore.update() | 无工作副本 | "无可用工作副本" |
| 更新到版本 | ArrowUpCircle | 打开 UpdateRevisionDialog | 无工作副本 | "无可用工作副本" |
| 日志 | History | router.push('/workspace/log') | 无工作副本 | "无可用工作副本" |
| 切换 | GitBranch | 打开 SwitchDialog | 无工作副本 | "无可用工作副本" |
| 分支/标签 | Layers | 打开 BranchTagDialog | 无工作副本 | "无可用工作副本" |
| 合并 | GitMerge | 打开 MergeWizard | 无工作副本 | "无可用工作副本" |
| 清理 | Broom | svnStore.cleanup() | 无工作副本 | "无可用工作副本" |
| 导出 | Package | 打开 ExportDialog | 无工作副本 | "无可用工作副本" |
| ⚙️ 设置 | Settings | router.push('/settings') | 始终可用 | — |

**溢出策略：** ResizeObserver 监听容器宽度，每个按钮约 88px，至少保留"提交"和"设置"可见。
**禁用态规则（三层判断）：**
1. `isOperationRunning=true` → 全部禁用，tooltip="操作进行中，请等待"（最高优先级）
2. 无工作副本 → 全部禁用（设置除外），tooltip="无可用工作副本"
3. 提交按钮：无工作副本 + 无变更时 → tooltip="当前工作副本无变更"

**核心代码（全组件骨架，使用 `lucide-vue-next` 组件模式）：**

```vue
<script setup lang="ts">
import { ref, computed, onMounted, onUnmounted, type Component } from 'vue'
import { useRouter } from 'vue-router'
import { GitCommit, RefreshCw, ArrowUpCircle, History, GitBranch, Layers, GitMerge, Broom, Package, Settings, MoreHorizontal } from 'lucide-vue-next'
import { useWorkspaceStore } from '@/stores/workspace'
import { useFileListStore } from '@/stores/fileList'

const router = useRouter()
const workspaceStore = useWorkspaceStore()
const fileListStore = useFileListStore()

const toolbarRef = ref<HTMLElement | null>(null)
const overflowCount = ref(0)

// lucide-vue-next 图标组件映射（用于动态渲染）
const iconMap: Record<string, Component> = {
  GitCommit, RefreshCw, ArrowUpCircle, History,
  GitBranch, Layers, GitMerge, Broom, Package,
  Settings, MoreHorizontal,
}

interface ToolbarButton {
  key: string; label: string; iconKey: string      // iconKey 对应 iconMap 中的键名
  action: () => void
  priority: number
  getDisabledTooltip: () => string                  // 计算禁用 tooltip
}

const wpUnavailable = computed(() => !workspaceStore.currentPath)
const globallyDisabled = computed(() => fileListStore.isOperationRunning)

const buttons = computed<ToolbarButton[]>(() => [
  {
    key: 'commit', label: '提交', iconKey: 'GitCommit',
    action: () => router.push('/workspace/commit'), priority: 8,
    getDisabledTooltip: () => {
      if (globallyDisabled.value) return '操作进行中，请等待'
      if (wpUnavailable.value) return '无可用工作副本'
      // 有工作副本但无变更时(预留 fileListStore 状态)
      return fileListStore.files.length === 0 ? '当前工作副本无变更' : '无可用工作副本'
    },
  },
  { key: 'update', label: '更新', iconKey: 'RefreshCw', action: () => {}, priority: 7,
    getDisabledTooltip: () => globallyDisabled.value ? '操作进行中，请等待' : '无可用工作副本' },
  { key: 'updateToRevision', label: '更新到版本', iconKey: 'ArrowUpCircle', action: () => {}, priority: 6,
    getDisabledTooltip: () => globallyDisabled.value ? '操作进行中，请等待' : '无可用工作副本' },
  { key: 'log', label: '日志', iconKey: 'History', action: () => router.push('/workspace/log'), priority: 6,
    getDisabledTooltip: () => globallyDisabled.value ? '操作进行中，请等待' : '无可用工作副本' },
  { key: 'switch', label: '切换', iconKey: 'GitBranch', action: () => {}, priority: 5,
    getDisabledTooltip: () => globallyDisabled.value ? '操作进行中，请等待' : '无可用工作副本' },
  { key: 'branchTag', label: '分支/标签', iconKey: 'Layers', action: () => {}, priority: 4,
    getDisabledTooltip: () => globallyDisabled.value ? '操作进行中，请等待' : '无可用工作副本' },
  { key: 'merge', label: '合并', iconKey: 'GitMerge', action: () => {}, priority: 3,
    getDisabledTooltip: () => globallyDisabled.value ? '操作进行中，请等待' : '无可用工作副本' },
  { key: 'cleanup', label: '清理', iconKey: 'Broom', action: () => {}, priority: 1,
    getDisabledTooltip: () => globallyDisabled.value ? '操作进行中，请等待' : '无可用工作副本' },
  { key: 'export', label: '导出', iconKey: 'Package', action: () => {}, priority: 0,
    getDisabledTooltip: () => globallyDisabled.value ? '操作进行中，请等待' : '无可用工作副本' },
])

const btnDisabled = computed(() => wpUnavailable.value || globallyDisabled.value)

const visibleButtons = computed(() => {
  const count = Math.max(1, buttons.value.length - overflowCount.value)
  return buttons.value.slice(0, count)
})
const overflowButtons = computed(() => buttons.value.slice(visibleButtons.value.length))

onMounted(() => {
  if (!toolbarRef.value) return
  const observer = new ResizeObserver(([entry]) => {
    const avail = entry.contentRect.width - 80
    overflowCount.value = Math.max(0, buttons.value.length - Math.max(1, Math.floor(avail / 88)))
  })
  observer.observe(toolbarRef.value)
  onUnmounted(() => observer.disconnect())
})
</script>

<template>
  <div ref="toolbarRef" class="h-10 px-3 flex items-center bg-white dark:bg-slate-800 border-b border-slate-200 dark:border-slate-700">
    <div class="flex items-center gap-1 flex-1 overflow-hidden">
      <template v-for="btn in visibleButtons" :key="btn.key">
        <el-tooltip
          :content="btn.getDisabledTooltip()"
          :disabled="!btnDisabled"
          effect="dark"
          placement="bottom"
        >
          <button
            class="inline-flex items-center gap-1 px-2.5 py-1 text-xs rounded-md whitespace-nowrap"
            :class="btnDisabled
              ? 'opacity-50 cursor-not-allowed text-slate-400 dark:text-slate-500'
              : 'text-slate-600 dark:text-slate-300 hover:bg-slate-100 dark:hover:bg-slate-700 hover:text-green-600 dark:hover:text-green-400'"
            :disabled="btnDisabled"
            :aria-label="btn.label"
            @click="btn.action"
          >
            <component :is="iconMap[btn.iconKey as keyof typeof iconMap]" class="w-4 h-4" /><span>{{ btn.label }}</span>
          </button>
        </el-tooltip>
      </template>
      <el-dropdown v-if="overflowButtons.length > 0" trigger="click" placement="bottom-end">
        <button class="px-1.5 py-1 text-xs text-slate-400" aria-label="更多操作">
          <MoreHorizontal class="w-4 h-4" />
        </button>
        <template #dropdown>
          <el-dropdown-menu>
            <el-dropdown-item v-for="btn in overflowButtons" :key="btn.key" :disabled="btnDisabled" @click="btn.action">
              <component :is="iconMap[btn.iconKey as keyof typeof iconMap]" class="w-4 h-4 mr-2" />{{ btn.label }}
            </el-dropdown-item>
          </el-dropdown-menu>
        </template>
      </el-dropdown>
    </div>
    <el-tooltip content="设置" placement="bottom">
      <button class="p-1.5 rounded-md text-slate-400 hover:text-slate-600 dark:hover:text-slate-300 hover:bg-slate-100 dark:hover:bg-slate-700" aria-label="设置" @click="router.push('/settings')">
        <Settings class="w-4 h-4" />
      </button>
    </el-tooltip>
  </div>
</template>
```
