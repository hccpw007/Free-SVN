## 操作细节

### 5.3A — 新建 `src/components/common/ToolBar.vue`

高度 40px（h-10），背景 `bg-white dark:bg-slate-800`，底部 1px 边框。使用 `<script setup lang="ts">` 语法，Lucide 图标（无 emoji）。

**完整按钮列表（10 个）：**

| 标签 | Lucide 图标 | 操作 | 禁用条件 |
|------|------------|------|---------|
| 提交 | git-commit | router.push('/workspace/commit') | 无工作副本 |
| 更新 | refresh-cw | svnStore.update() | 无工作副本 |
| 更新到版本 | arrow-up-circle | 打开 UpdateRevisionDialog | 无工作副本 |
| 日志 | history | router.push('/workspace/log') | 无工作副本 |
| 切换 | git-branch | 打开 SwitchDialog | 无工作副本 |
| 分支/标签 | layers | 打开 BranchTagDialog | 无工作副本 |
| 合并 | git-merge | 打开 MergeWizard | 无工作副本 |
| 清理 | broom | svnStore.cleanup() | 无工作副本 |
| 导出 | package | 打开 ExportDialog | 无工作副本 |
| ⚙️ 设置 | settings | router.push('/settings') | 始终可用 |

**溢出策略：** ResizeObserver 监听容器宽度，每个按钮约 88px，至少保留"提交"和"设置"可见。
**禁用态：** `isOperationRunning` 时全部禁用（`opacity-50 cursor-not-allowed`），tooltip="操作进行中，请等待"。

**核心代码（全组件骨架）：**

```vue
<script setup lang="ts">
import { ref, computed, onMounted, onUnmounted } from 'vue'
import { useRouter } from 'vue-router'
import { useWorkspaceStore } from '@/stores/workspace'
import { useFileListStore } from '@/stores/fileList'

const router = useRouter()
const workspaceStore = useWorkspaceStore()
const fileListStore = useFileListStore()

const toolbarRef = ref<HTMLElement | null>(null)
const overflowCount = ref(0)

interface ToolbarButton {
  key: string; label: string; icon: string
  action: () => void
  priority: number  // 越大越优先保留
}

const wpUnavailable = computed(() => !workspaceStore.currentPath)

const buttons = computed<ToolbarButton[]>(() => [
  { key: 'commit', label: '提交', icon: 'git-commit', action: () => router.push('/workspace/commit'), priority: 8 },
  { key: 'update', label: '更新', icon: 'refresh-cw', action: () => {}, priority: 7 },
  { key: 'updateToRevision', label: '更新到版本', icon: 'arrow-up-circle', action: () => {}, priority: 2 },
  { key: 'log', label: '日志', icon: 'history', action: () => router.push('/workspace/log'), priority: 6 },
  { key: 'switch', label: '切换', icon: 'git-branch', action: () => {}, priority: 5 },
  { key: 'branchTag', label: '分支/标签', icon: 'layers', action: () => {}, priority: 4 },
  { key: 'merge', label: '合并', icon: 'git-merge', action: () => {}, priority: 3 },
  { key: 'cleanup', label: '清理', icon: 'broom', action: () => {}, priority: 1 },
  { key: 'export', label: '导出', icon: 'package', action: () => {}, priority: 0 },
])

const visibleButtons = computed(() => {
  const count = Math.max(1, buttons.value.length - overflowCount.value)
  return buttons.value.slice(0, count)
})
const overflowButtons = computed(() => buttons.value.slice(visibleButtons.value.length))

onMounted(() => {
  if (!toolbarRef.value) return
  const observer = new ResizeObserver(([entry]) => {
    const avail = entry.contentRect.width - 80
    overflowCount.value = Math.max(0, buttons.value.length - Math.max(1, Math.floor(avail / 88)))
  })
  observer.observe(toolbarRef.value)
  onUnmounted(() => observer.disconnect())
})

const globallyDisabled = computed(() => fileListStore.isOperationRunning)
const btnDisabled = computed(() => wpUnavailable.value || globallyDisabled.value)
</script>

<template>
  <div ref="toolbarRef" class="h-10 px-3 flex items-center bg-white dark:bg-slate-800 border-b border-slate-200 dark:border-slate-700">
    <div class="flex items-center gap-1 flex-1 overflow-hidden">
      <template v-for="btn in visibleButtons" :key="btn.key">
        <el-tooltip :content="globallyDisabled ? '操作进行中，请等待' : btn.label" effect="dark" placement="bottom">
          <button
            class="inline-flex items-center gap-1 px-2.5 py-1 text-xs rounded-md whitespace-nowrap"
            :class="btnDisabled ? 'opacity-50 cursor-not-allowed text-slate-400' : 'text-slate-600 hover:bg-slate-100 hover:text-green-600 dark:text-slate-300 dark:hover:bg-slate-700 dark:hover:text-green-400'"
            :disabled="btnDisabled"
            :aria-label="btn.label"
            @click="btn.action"
          >
            <svg class="w-4 h-4" :data-lucide="btn.icon" /><span>{{ btn.label }}</span>
          </button>
        </el-tooltip>
      </template>
      <el-dropdown v-if="overflowButtons.length > 0" trigger="click" placement="bottom-end">
        <button class="px-1.5 py-1 text-xs text-slate-400"><svg class="w-4 h-4" data-lucide="more-horizontal" /></button>
        <template #dropdown>
          <el-dropdown-menu>
            <el-dropdown-item v-for="btn in overflowButtons" :key="btn.key" :disabled="btnDisabled" @click="btn.action">
              <svg class="w-4 h-4 mr-2" :data-lucide="btn.icon" />{{ btn.label }}
            </el-dropdown-item>
          </el-dropdown-menu>
        </template>
      </el-dropdown>
    </div>
    <el-tooltip content="设置" placement="bottom">
      <button class="p-1.5 rounded-md text-slate-400 hover:text-slate-600 dark:hover:text-slate-300 hover:bg-slate-100 dark:hover:bg-slate-700" aria-label="设置" @click="router.push('/settings')">
        <svg class="w-4 h-4" data-lucide="settings" />
      </button>
    </el-tooltip>
  </div>
</template>
```
