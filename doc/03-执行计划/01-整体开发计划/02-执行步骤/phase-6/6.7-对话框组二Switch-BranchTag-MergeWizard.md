### 6.7A — SwitchDialog

el-dialog 460px。当前路径只读 + 目标路径下拉（EL-Select filterable allow-create，prepend `^/branches` 分支列表）+ 手动输入 + ignoreAncestry 复选框。切换中按钮 loading。

**流程：** `svnStore.listBranches()` → 下拉选单 → `svnStore.switchBranch()` → toast + 关闭。

**完整代码：**

```vue
<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { useWorkspaceStore } from '@/stores/workspace'
import { useSvnStore } from '@/stores/svn'
import { useI18n } from 'vue-i18n'
import { ElMessage } from 'element-plus'

const { t } = useI18n()
const workspaceStore = useWorkspaceStore()
const svnStore = useSvnStore()

const emit = defineEmits<{ close: [] }>()

const branches = ref<string[]>([])
const targetUrl = ref('')
const ignoreAncestry = ref(false)
const isSwitching = ref(false)

onMounted(async () => {
  // 用户手动输入分支路径，不调用后端 listBranches（需要后端 svn ls ^/branches command）
  // 如后续添加 listBranches 后端 command，在此处调用
  try {
    const { invoke } = await import('@tauri-apps/api/core')
    branches.value = await invoke<string[]>('list_branches', { url: workspaceStore.url })
  } catch { /* 忽略：用户仍可手动输入分支路径 */ }
})

async function handleSwitch() {
  if (!targetUrl.value || !workspaceStore.currentPath) return
  isSwitching.value = true
  try {
    await svnStore.switchBranch({ path: workspaceStore.currentPath, targetUrl: targetUrl.value, ignoreAncestry: ignoreAncestry.value })
    ElMessage.success(t('workspace.switchSuccess'))
    emit('close')
  } catch { ElMessage.error(t('workspace.switchFailed')) }
  finally { isSwitching.value = false }
}
</script>

<template>
  <el-dialog :model-value="true" :title="t('workspace.switchTitle')" width="460px" :close-on-click-modal="false" @close="emit('close')">
    <div class="space-y-3">
      <div>
        <label class="text-xs text-slate-500 dark:text-slate-400">{{ t('workspace.switchCurrentPath') }}</label>
        <p class="text-xs font-mono text-slate-700 dark:text-slate-300 mt-1">{{ workspaceStore.currentPath }}</p>
      </div>
      <div>
        <label class="text-xs text-slate-500 dark:text-slate-400">{{ t('workspace.switchTarget') }}</label>
        <el-select v-model="targetUrl" filterable allow-create size="default" class="!w-full mt-1" placeholder="选择或输入分支路径...">
          <el-option v-for="b in branches" :key="b" :label="b" :value="b" />
        </el-select>
      </div>
      <el-checkbox v-model="ignoreAncestry"><span class="text-xs">{{ t('workspace.ignoreAncestry') }}</span></el-checkbox>
    </div>
    <template #footer>
      <el-button size="default" @click="emit('close')">{{ t('common.cancel') }}</el-button>
      <el-button size="default" type="primary" :loading="isSwitching" :disabled="!targetUrl" @click="handleSwitch">{{ t('workspace.switch') }}</el-button>
    </template>
  </el-dialog>
</template>
```

### 6.7B — BranchTagDialog

el-dialog 460px。来源 URL 自动填充（显示当前工作副本 URL）→ 目标路径输入（placeholder 提示 branches/tags 下）→ 日志 textarea → 版本号可选 ElInputNumber → [创建] 按钮。

**流程：** `svnStore.copyBranchTag()` → toast + 关闭。

**完整代码：**

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { useWorkspaceStore } from '@/stores/workspace'
import { useSvnStore } from '@/stores/svn'
import { useI18n } from 'vue-i18n'
import { ElMessage } from 'element-plus'

const { t } = useI18n()
const workspaceStore = useWorkspaceStore()
const svnStore = useSvnStore()

const emit = defineEmits<{ close: [] }>()

/* workspaceStore.url 通过 getInfo 填充，在 switchWorkspace 或 onMounted 中调用 fetchInfo 后设置 */
const srcUrl = computed(() => workspaceStore.sourceUrl || workspaceStore.currentPath)
const targetUrl = ref('')
const logMessage = ref('')
const revision = ref<number | undefined>(undefined)
const isCreating = ref(false)

async function handleCreate() {
  if (!targetUrl.value) return
  isCreating.value = true
  try {
    await svnStore.copyBranchTag({ srcUrl: srcUrl.value, dstUrl: targetUrl.value, message: logMessage.value || t('workspace.branchDefaultMessage'), revision: revision.value })
    ElMessage.success(t('workspace.branchCreated') || '创建成功')
    emit('close')
  } catch { ElMessage.error(t('workspace.branchFailed') || '创建失败') }
  finally { isCreating.value = false }
}
</script>

<template>
  <el-dialog :model-value="true" :title="t('workspace.branchTagTitle')" width="460px" :close-on-click-modal="false" @close="emit('close')">
    <div class="space-y-3">
      <div>
        <label class="text-xs text-slate-500 dark:text-slate-400">{{ t('workspace.sourceUrl') }}</label>
        <!-- workspaceStore.sourceUrl 通过 getInfo 填充，在 switchWorkspace 时自动更新 -->
        <p class="text-xs font-mono text-slate-700 dark:text-slate-300 mt-1">{{ srcUrl }}</p>
      </div>
      <div>
        <label class="text-xs text-slate-500 dark:text-slate-400">{{ t('workspace.targetUrl') }}</label>
        <el-input v-model="targetUrl" size="default" class="mt-1" :placeholder="t('workspace.targetUrlHint') || 'svn://.../branches/xxx（路径应在 branches/ 或 tags/ 下）'" />
      </div>
      <div>
        <label class="text-xs text-slate-500 dark:text-slate-400">{{ t('common.logMessage') }}</label>
        <el-input v-model="logMessage" type="textarea" :rows="3" size="default" class="mt-1" />
      </div>
      <div>
        <label class="text-xs text-slate-500 dark:text-slate-400">{{ t('common.revision') }}</label>
        <el-input-number v-model="revision" :min="1" size="default" class="!w-32 mt-1" :placeholder="t('common.optional') || '留空为 HEAD'" />
      </div>
    </div>
    <template #footer>
      <el-button size="default" @click="emit('close')">{{ t('common.cancel') }}</el-button>
      <el-button size="default" type="primary" :loading="isCreating" :disabled="!targetUrl" @click="handleCreate">{{ t('common.create') }}</el-button>
    </template>
  </el-dialog>
</template>
```

### 6.7C — MergeWizard（4 步向导）

el-dialog 520px，`close-on-click-modal=false`。步骤进度条在顶部（①→②→③→④，完成后 ✅ 标记）。

| 步骤 | 内容 | 按钮 |
|------|------|------|
| 1: 选择来源 | 源 URL 输入 | [下一步] |
| 2: 版本范围 | 起始/结束版本号（ElInputNumber） | [上一步] [下一步] |
| 3: 选项 | 3 个 checkbox（ignoreAncestry/recordOnly/allowMixed） | [上一步] [下一步] |
| 4: 执行 | 摘要信息 + [执行合并] + 结果面板 | [上一步] [完成] |

**结果三种状态：** `success`→绿色"合并完成，无冲突"；`conflict`→琥珀色"N 个文件存在冲突"；`error`→红色错误信息。

**完整代码：**

```vue
<script setup lang="ts">
import { ref, computed } from 'vue'
import { useWorkspaceStore } from '@/stores/workspace'
import { useSvnStore } from '@/stores/svn'
import { useI18n } from 'vue-i18n'

const { t } = useI18n()
const workspaceStore = useWorkspaceStore()
const svnStore = useSvnStore()

const emit = defineEmits<{ close: [] }>()

const currentStep = ref(1)
const totalSteps = 4

// 步骤 1: 选择来源
const sourceUrl = ref('')
// 步骤 2: 版本范围
const revStart = ref<number | undefined>(undefined)
const revEnd = ref<number | undefined>(undefined)
// 步骤 3: 合并选项
const ignoreAncestry = ref(false)
const recordOnly = ref(false)
const allowMixedRevisions = ref(false)
// 步骤 4: 执行与结果
const isMerging = ref(false)
type ResultType = 'idle' | 'success' | 'conflict' | 'error'
const resultStatus = ref<ResultType>('idle')
const resultMessage = ref('')

const stepStatus = computed(() => [
  currentStep.value > 1 ? 'success' : 'process',
  currentStep.value > 2 ? 'success' : currentStep.value === 2 ? 'process' : 'wait',
  currentStep.value > 3 ? 'success' : currentStep.value === 3 ? 'process' : 'wait',
  resultStatus.value !== 'idle' ? 'success' : currentStep.value === 4 ? 'process' : 'wait',
])

function nextStep() { if (currentStep.value < totalSteps) currentStep.value++ }
function prevStep() { if (currentStep.value > 1) currentStep.value-- }

async function handleMerge() {
  if (!sourceUrl.value || !workspaceStore.currentPath) return
  isMerging.value = true; resultStatus.value = 'idle'
  try {
    const r = await svnStore.mergeBranch({ srcUrl: sourceUrl.value, revStart: revStart.value, revEnd: revEnd.value, targetPath: workspaceStore.currentPath })
    if (r.conflictCount && r.conflictCount > 0) { resultStatus.value = 'conflict'; resultMessage.value = `${r.conflictCount} 个文件存在冲突` }
    else { resultStatus.value = 'success'; resultMessage.value = '合并完成，无冲突' }
  } catch (e: any) { resultStatus.value = 'error'; resultMessage.value = e.message || '合并失败' }
  finally { isMerging.value = false }
}
</script>

<template>
  <el-dialog :model-value="true" :title="t('workspace.mergeTitle')" width="520px" :close-on-click-modal="false" @close="emit('close')">
    <el-steps :active="currentStep" align-center class="mb-6" :space="120">
      <el-step v-for="(title, i) in [t('workspace.mergeStep1Title'), t('workspace.mergeStep2Title'), t('workspace.mergeStep3Title'), t('workspace.mergeStep4Title')]" :key="i"
        :title="title" :status="stepStatus[i]" />
    </el-steps>

    <!-- 步骤 1 -->
    <div v-if="currentStep === 1" class="space-y-3">
      <p class="text-sm font-medium text-slate-700 dark:text-slate-300">{{ t('workspace.mergeStep1') }}</p>
      <el-input v-model="sourceUrl" size="default" :placeholder="t('workspace.sourceUrl') || '源 URL'" />
    </div>

    <!-- 步骤 2 -->
    <div v-if="currentStep === 2" class="space-y-3">
      <p class="text-sm font-medium text-slate-700 dark:text-slate-300">{{ t('workspace.mergeStep2') }}</p>
      <div class="flex gap-3">
        <div><label class="text-xs text-slate-500">{{ t('workspace.revStart') || '起始版本' }}</label>
          <el-input-number v-model="revStart" :min="1" size="default" class="!w-32 mt-1" :placeholder="t('common.optional')" /></div>
        <div><label class="text-xs text-slate-500">{{ t('workspace.revEnd') || '结束版本' }}</label>
          <el-input-number v-model="revEnd" :min="1" size="default" class="!w-32 mt-1" :placeholder="t('common.optional')" /></div>
      </div>
    </div>

    <!-- 步骤 3 -->
    <div v-if="currentStep === 3" class="space-y-3">
      <p class="text-sm font-medium text-slate-700 dark:text-slate-300">{{ t('workspace.mergeStep3') }}</p>
      <div class="flex flex-col gap-2">
        <el-checkbox v-model="ignoreAncestry"><span class="text-xs">{{ t('workspace.ignoreAncestry') }}</span></el-checkbox>
        <el-checkbox v-model="recordOnly"><span class="text-xs">{{ t('workspace.recordOnly') }}</span></el-checkbox>
        <el-checkbox v-model="allowMixedRevisions"><span class="text-xs">{{ t('workspace.allowMixed') }}</span></el-checkbox>
      </div>
    </div>

    <!-- 步骤 4 -->
    <div v-if="currentStep === 4" class="space-y-3">
      <p class="text-sm font-medium text-slate-700 dark:text-slate-300">{{ t('workspace.mergeStep4') }}</p>
      <div class="bg-slate-50 dark:bg-slate-900 rounded-md p-3 text-xs space-y-1">
        <p class="text-slate-500">目标: {{ workspaceStore.currentPath }}</p>
        <p class="text-slate-500">来源: {{ sourceUrl }}</p>
      </div>
      <el-button size="default" type="primary" :loading="isMerging" :disabled="!sourceUrl" @click="handleMerge">{{ t('workspace.executeMerge') }}</el-button>
      <div v-if="resultStatus !== 'idle'" class="rounded-md p-3 text-xs"
        :class="resultStatus === 'success' ? 'bg-green-50 text-green-700 dark:bg-green-900/20 dark:text-green-400' : resultStatus === 'conflict' ? 'bg-amber-50 text-amber-700 dark:bg-amber-900/20 dark:text-amber-400' : 'bg-red-50 text-red-700 dark:bg-red-900/20 dark:text-red-400'">
        {{ resultMessage }}
      </div>
    </div>

    <template #footer>
      <div class="flex justify-between">
        <el-button size="default" @click="emit('close')">{{ t('common.cancel') }}</el-button>
        <div class="flex gap-2">
          <el-button v-if="currentStep > 1" size="default" @click="prevStep">{{ t('common.prev') }}</el-button>
          <el-button v-if="currentStep < totalSteps" size="default" type="primary" @click="nextStep">{{ t('common.next') }}</el-button>
          <el-button v-if="currentStep === totalSteps && resultStatus !== 'idle'" size="default" type="primary" @click="emit('close')">{{ t('common.finish') }}</el-button>
        </div>
      </div>
    </template>
  </el-dialog>
</template>
```
