### 6.2A — 修改 `src/components/svn/FileListTable.vue`

Element Plus 表格 `<el-table size="small">`，行高 40px。四列：复选框 + 状态图标 + 文件名 + 操作。底栏固定在表格外部（shrink-0）。右键菜单通过 Teleport + 自定义 div 实现。

```vue
<script setup lang="ts">
import { ref } from 'vue'
import { useFileListStore } from '@/stores/fileList'
import { useI18n } from 'vue-i18n'
import { useWorkspaceStore } from '@/stores/workspace'
import { open } from '@tauri-apps/plugin-shell'
import { join } from '@tauri-apps/api/path'
import FileStatusIcon from './FileStatusIcon.vue'
import FileActionButtons from './FileActionButtons.vue'

const { t } = useI18n()
const fileListStore = useFileListStore()
const workspaceStore = useWorkspaceStore()

// 右键菜单状态
const ctxMenu = ref({ show: false, x: 0, y: 0, file: null as any })

// 冲突行红色指示条：通过 row-class-name 回调为冲突行添加 Tailwind class
// el-table 的 row-class-name 回调签名：(data: { row: any, rowIndex: number }) => string
function rowClassName({ row }: { row: { status: string } }): string {
  return row.status === 'conflicted' ? 'border-l-[3px] border-red-500' : ''
}

function handleRowClick(row: any, _column: any, event: MouseEvent) {
  const set = new Set(fileListStore.selectedPaths)
  if (event.shiftKey) {
    // Shift+Click: 范围选择
    const allFiles = fileListStore.filteredFiles
    const idx = allFiles.findIndex(f => f.path === row.path)
    if (idx >= 0) {
      for (let i = 0; i <= idx; i++) set.add(allFiles[i].path)
    }
  } else if (event.ctrlKey || event.metaKey) {
    // Ctrl+Click: 切换选择
    set.has(row.path) ? set.delete(row.path) : set.add(row.path)
  } else {
    // 普通点击: 单选
    set.clear()
    set.add(row.path)
  }
  fileListStore.selectedPaths = set
}

function handleContextMenu(e: MouseEvent, row: any) {
  e.preventDefault()
  ctxMenu.value = { show: true, x: e.clientX, y: e.clientY, file: row }
}

async function copyFullPath(p: string) {
  // 使用 Tauri path.join 确保跨平台路径拼接（避免 Windows 混合分隔符问题）
  const absolutePath = await join(workspaceStore.currentPath, p)
  try { await navigator.clipboard.writeText(absolutePath) } catch { /* fallback */ }
}

async function copyRelativePath(p: string) {
  try { await navigator.clipboard.writeText(p) } catch { /* fallback */ }
}

async function showInExplorer(p: string) {
  const fullPath = await join(workspaceStore.currentPath, p)
  try { await open(fullPath) } catch { /* fallback */ }
}

async function openWithEditor(p: string) {
  const fullPath = await join(workspaceStore.currentPath, p)
  try { await open(fullPath) } catch { /* fallback */ }
}

// 事件冒泡架构（三层结构）：
// 第一层 FileActionButtons: 通过 emit 发出 diff/revert/ignore/delete/merge/unlock 事件
// 第二层 FileListTable: 接收 FileActionButtons 的 emit，根据操作类型决定处理方式
//   - diff → 通过 emit 向上转发给 HomePage（HomePage 监听后打开差异对比面板）
//   - revert/ignore/delete/unlock → 在此处调用对应的 store action 执行
//   - merge → 通过 emit 向上转发给 HomePage（后续实现 MergeWizard 对话框）
// 第三层 HomePage: 监听 FileListTable 发出的 diff/merge 事件，打开对话框或面板
defineEmits<{ diff: [path: string]; merge: [path: string] }>()
function diffFile(path: string) { /* 触发 emit('diff', path) → HomePage 监听 */ }
function revertFile(path: string) { /* fileListStore.revertFile(path) 内部处理 */ }
function ignoreFile(path: string) { /* fileListStore.ignoreFile(path) 内部处理 */ }
function deleteFile(path: string) { /* fileListStore.deleteFile(path) 内部处理 */ }
function mergeFile(path: string) { /* 触发 emit('merge', path) → HomePage 监听 */ }
function unlockFile(path: string) { /* fileListStore.unlockFile(path) 内部处理 */ }
</script>

<template>
  <div class="h-full flex flex-col">
    <el-table
      :data="fileListStore.filteredFiles"
      size="small" style="width:100%;height:100%" row-key="path"
      :row-class-name="rowClassName"
      @row-click="handleRowClick"
      @cell-contextmenu="(r: any) => handleContextMenu($event, r)"
      v-loading="fileListStore.isLoading"
    >
      <!-- 注意：type="selection" 与自定义 selectedPaths 管理存在状态冲突风险；
	     使用自定义复选框列替代 el-table 内置 selection -->
	      <el-table-column width="40">
	        <template #default="{ row }">
	          <el-checkbox :model-value="fileListStore.selectedPaths.has(row.path)" @change="(v: boolean) => fileListStore.toggleSelect(row.path)" />
	        </template>
	      </el-table-column>
      <el-table-column :label="t('file.status')" width="48" sortable="custom" prop="status">
        <template #default="{ row }"><FileStatusIcon :status="row.status" /></template>
      </el-table-column>
      <el-table-column :label="t('file.fileName')" min-width="200" sortable="custom" prop="path">
        <template #default="{ row }">
          <span class="truncate block text-sm font-mono text-slate-700 dark:text-slate-300" :title="row.path">{{ row.path }}</span>
        </template>
      </el-table-column>
      <el-table-column :label="t('file.actions')" width="180" fixed="right">
        <template #default="{ row }"><FileActionButtons :file="row" @diff="diffFile" @revert="revertFile" @ignore="ignoreFile" @delete="deleteFile" @merge="mergeFile" @unlock="unlockFile" /></template>
      </el-table-column>
    </el-table>

    <!-- 固定底栏 -->
    <div class="shrink-0 h-9 px-4 flex items-center gap-3 text-xs border-t border-slate-200 dark:border-slate-700 bg-white dark:bg-slate-800">
      <el-checkbox
        :model-value="fileListStore.selectedPaths.size === fileListStore.filteredFiles.length && fileListStore.filteredFiles.length > 0"
        :indeterminate="fileListStore.selectedPaths.size > 0 && fileListStore.selectedPaths.size < fileListStore.filteredFiles.length"
        @change="(v: boolean) => fileListStore.toggleSelectAll(v)"
      />
      <span class="text-slate-500 dark:text-slate-400">{{ t('workspace.selectedCount', { count: fileListStore.selectedPaths.size }) }}</span>
      <el-button v-if="fileListStore.selectedPaths.size > 0" size="small" @click="/* 批量还原 */">
        {{ t('file.batchRevert') }}
      </el-button>
    </div>

    <!-- 右键菜单 -->
    <Teleport to="body">
      <div v-if="ctxMenu.show" class="fixed z-50 w-48 py-1 rounded-md bg-white dark:bg-slate-800 shadow-lg border border-slate-200 dark:border-slate-700 text-xs transition-all duration-200 ease-out"
        :style="{ left: ctxMenu.x + 'px', top: ctxMenu.y + 'px' }" @click.stop
      >
        <button class="w-full px-3 py-1.5 text-left text-slate-700 hover:bg-slate-100 dark:text-slate-300 dark:hover:bg-slate-700" @click="copyFullPath(ctxMenu.file.path); ctxMenu.show = false">{{ t('file.copyFullPath') }}</button>
        <button class="w-full px-3 py-1.5 text-left text-slate-700 hover:bg-slate-100 dark:text-slate-300 dark:hover:bg-slate-700" @click="copyRelativePath(ctxMenu.file.path); ctxMenu.show = false">{{ t('file.copyRelativePath') }}</button>
        <div class="border-t border-slate-200 dark:border-slate-700 my-1" />
        <button class="w-full px-3 py-1.5 text-left text-slate-700 hover:bg-slate-100 dark:text-slate-300 dark:hover:bg-slate-700" @click="showInExplorer(ctxMenu.file.path); ctxMenu.show = false">{{ t('file.showInExplorer') }}</button>
        <button class="w-full px-3 py-1.5 text-left text-slate-700 hover:bg-slate-100 dark:text-slate-300 dark:hover:bg-slate-700" @click="openWithEditor(ctxMenu.file.path); ctxMenu.show = false">{{ t('file.openWithEditor') }}</button>
      </div>
      <!-- 关闭层 -->
      <div v-if="ctxMenu.show" class="fixed inset-0 z-40" @click="ctxMenu.show = false" />
    </Teleport>
  </div>
</template>
```

**默认排序逻辑：** 排序优先级 conflicted > modified > added > deleted > unversioned > locked，各状态内按文件名升序。表头点击切换升序/降序。

**排序实现细节（在 FileListTable 中通过 `sort-change` 事件更新 fileListStore）：**

```vue
// FileListTable.vue — sort-change 事件处理
<el-table @sort-change="({ prop, order }: any) => {
  fileListStore.sortField = prop || 'status'
  fileListStore.sortOrder = order === 'ascending' ? 'asc' : 'desc'
}">
```

**多选交互实现细节（handleRowClick 完整逻辑）：**

el-table 的 `@row-click` 事件传递 `(row, column, event)`。普通点击设置单选，Ctrl/Cmd+Click 切换选择状态，Shift+Click 从上次选中的文件到当前文件之间的所有文件全部选中。`selectedPaths` 使用 `Set<string>` 存储，避免重复选中。

```ts
// 完整的多选交互逻辑已在上面 handleRowClick 中实现
// 要点：
// - Shift+Click：范围选择（从 lastIdx 到当前 idx）

// - 普通点击：清除之前选中，仅选中当前文件
// - lastIdx 在切换工作副本或清除选择时重置
```

**右键菜单行为逻辑（业务设计 3.2）：**
- 工作副本目录右键  显示 12 项 SVN 菜单（提交/更新/日志/差异/还原/切换/分支标签/合并/清理/更新到版本/导出）
- 普通文件夹右键  仅 SVN 检出
- 文件右键  显示 12 项 SVN 菜单（同上）
- 右键菜单优先级：Shell 集成菜单（7.4 实现）与表格内部右键菜单（6.2 实现）分离。Shell 集成处理全系统级右键操作，表格内部右键处理文件行快捷操作（复制路径/打开文件）

**表格内部右键菜单实现细节（Teleport + 自定义 div）：**

右键菜单通过 Teleport to body 的自定义 div 实现，4 项操作：复制文件路径/复制相对路径/在文件管理器中显示/用外部编辑器打开。菜单定位使用 `e.clientX` 和 `e.clientY`。点击菜单项后 `ctxMenu.show = false` 关闭菜单，点击菜单外部通过覆盖层 `@click` 关闭。菜单项 hover 背景色切换 `hover:bg-slate-100 dark:hover:bg-slate-700`。

**右键菜单 i18n key 说明：** 右键菜单 4 项对应 i18n key 已在 `en.json` / `zh-CN.json` 中定义：
- `file.copyFullPath` → "复制文件路径" / "Copy Full Path"
- `file.copyRelativePath` → "复制相对路径" / "Copy Relative Path"
- `file.showInExplorer` → "在文件管理器中显示" / "Show in Explorer"
- `file.openWithEditor` → "用外部编辑器打开" / "Open with Editor"

**el-table API 用法说明：**

| API | 用途 | 参数 |
|-----|------|------|
| `@row-click` | 行点击（多选交互） | (row, column, event) |
| `@cell-contextmenu` | 单元格右键菜单 | (row, column, event) |
| `@sort-change` | 表头排序变化 | ({ prop, order }) |
| `row-key` | 行唯一标识 | `"path"` |
| `type="selection"` | 复选框列 | width=40 |
| `sortable="custom"` | 自定义排序 | 配合 `@sort-change` 使用 |
| `v-loading` | 加载状态 | `fileListStore.isLoading` |
| `size="small"` | 紧凑模式 | 行高约 40px |
| `fixed="right"` | 固定列 | 操作列固定在右侧 |

### 6.2C — 新建 `src/components/svn/FileActionButtons.vue`

```vue
<script setup lang="ts">
import { computed } from 'vue'
import { useFileListStore } from '@/stores/fileList'
import { useI18n } from 'vue-i18n'

const { t } = useI18n()
const props = defineProps<{ file: { status: string; path: string } }>()
const emit = defineEmits<{ diff: [path: string]; revert: [path: string]; ignore: [path: string]; delete: [path: string]; merge: [path: string]; unlock: [path: string] }>()
const fileListStore = useFileListStore()
const disabled = computed(() => fileListStore.isOperationRunning)
const isOp = (s: string) => props.file.status === s

const btn = (labelKey: string, cls: string, action: string) => ({
  labelKey, cls, handler: () => emit(action as any, props.file.path)
})
</script>
<template>
  <div class="flex items-center gap-2">
    <button v-for="b in (
      isOp('modified') ? [btn('file.diff','text-blue-600','diff'),btn('file.revert','text-amber-600','revert')] :
      isOp('added') ? [btn('file.diff','text-blue-600','diff'),btn('file.ignore','text-slate-500','ignore'),btn('file.delete','text-red-500','delete')] :
      isOp('conflicted') ? [btn('file.diff','text-blue-600','diff'),btn('file.merge','text-red-600','merge')] :
      isOp('deleted') ? [btn('file.diff','text-blue-600','diff'),btn('file.ignore','text-slate-500','ignore'),btn('file.revert','text-amber-600','revert')] :
      isOp('unversioned') ? [btn('file.diff','text-blue-600','diff'),btn('file.ignore','text-slate-500','ignore')] :
      isOp('locked') ? [btn('file.diff','text-blue-600','diff'),btn('file.unlock','text-violet-600','unlock')] : []
    )" :key="b.labelKey" :disabled="disabled"
      class="text-xs underline-offset-2 hover:underline transition-colors duration-150 disabled:opacity-50 disabled:cursor-not-allowed"
      :class="[b.cls, disabled ? 'opacity-50' : '']" @click.stop="b.handler"
    >{{ t(b.labelKey) }}</button>
  </div>
</template>
```
