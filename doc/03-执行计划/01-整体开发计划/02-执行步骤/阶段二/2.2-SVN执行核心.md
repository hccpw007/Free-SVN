```rust
use crate::models::error::AppError;
use log;
use std::io::{BufRead, BufReader};
use std::path::PathBuf;
use std::process::{Command, Stdio};
use std::sync::atomic::{AtomicBool, Ordering};
use std::sync::Mutex;
use std::time::Instant;
use tokio::task::spawn_blocking;
use tokio::time::{timeout, Duration};

/// 全局取消标志
pub static CANCELLED: AtomicBool = AtomicBool::new(false);

/// 当前 svn 子进程句柄
pub static CURRENT_CHILD: Mutex<Option<std::process::Child>> = Mutex::new(None);

const DEFAULT_TIMEOUT_SECS: u64 = 60;
const BLAME_TIMEOUT_SECS: u64 = 120;

/// 认证提示关键词
const AUTH_KEYWORDS: &[&str] = &[
    "Authentication realm",
    "Password:",
    "Username:",
    "认证领域",
    "密码:",
    "用户名:",
];

/// 基础 SVN 参数（防止交互式认证对话框弹出）
const BASE_SVN_ARGS: &[&str] = &[
    "--non-interactive",
    "--trust-server-cert-failures",
    "unknown-ca,cn-mismatch,expired,not-yet-valid,other",
];

// ── 路径与版本 ──────────────────────────────────────────

pub fn get_svn_path() -> PathBuf {
    #[cfg(debug_assertions)]
    {
        PathBuf::from("svn")
    }
    #[cfg(not(debug_assertions))]
    {
        let resource_dir = tauri::api::path::resource_dir()
            .unwrap_or_else(|| PathBuf::from("."));
        #[cfg(target_os = "macos")]
        { resource_dir.join("resources/svn/macos/svn") }
        #[cfg(target_os = "windows")]
        { resource_dir.join("resources/svn/win/svn.exe") }
        #[cfg(target_os = "linux")]
        { resource_dir.join("resources/svn/linux/svn") }
    }
}

pub fn get_timeout_secs(args: &[&str]) -> u64 {
    if args.iter().any(|a| *a == "blame") {
        BLAME_TIMEOUT_SECS
    } else {
        DEFAULT_TIMEOUT_SECS
    }
}

pub fn validate_path(path: &str) -> Result<&str, AppError> {
    if path.is_empty() {
        return Err(AppError::InvalidInput("路径不能为空".into()));
    }
    if path.contains('\0') {
        return Err(AppError::InvalidInput("路径包含非法字符".into()));
    }
    Ok(path)
}

// ── 取消检测 ──────────────────────────────────────────

pub fn is_cancelled() -> bool {
    CANCELLED.load(Ordering::SeqCst)
}

pub fn set_cancelled(val: bool) {
    CANCELLED.store(val, Ordering::SeqCst);
}

// ── 认证检测 ──────────────────────────────────────────

fn check_auth_hint(stderr: &str) -> bool {
    AUTH_KEYWORDS.iter().any(|kw| stderr.contains(kw))
}

// ── 网络检测 ──────────────────────────────────────────

fn extract_host(url: &str) -> Result<String, AppError> {
    let without_scheme = url
        .strip_prefix("svn://")
        .or_else(|| url.strip_prefix("svn+ssh://"))
        .or_else(|| url.strip_prefix("https://"))
        .or_else(|| url.strip_prefix("http://"))
        .ok_or_else(|| AppError::InvalidInput(format!("无法识别的 URL 格式: {}", url)))?;

    let host = without_scheme
        .split('/')
        .next()
        .ok_or_else(|| AppError::InvalidInput(format!("URL 中未找到主机名: {}", url)))?;

    Ok(host.to_string())
}

/// 检测网络是否可达（5s 超时 TcpStream 连接尝试）
pub async fn check_network(server_url: &str) -> Result<(), AppError> {
    let host = extract_host(server_url)?;

    let ports: &[u16] = if server_url.starts_with("https://") {
        &[443]
    } else if server_url.starts_with("svn+ssh://") {
        &[22]
    } else {
        &[3690, 443]
    };

    for &port in ports {
        let addr = format!("{}:{}", host, port);
        match timeout(Duration::from_secs(5), tokio::net::TcpStream::connect(&addr)).await {
            Ok(Ok(_)) => return Ok(()),
            _ => continue,
        }
    }

    log::warn!("check_network 失败: host={}", host);
    Err(AppError::NetworkUnreachable)
}

// ── 核心执行 ──────────────────────────────────────────

fn run_svn_sync(svn_path: &PathBuf, args: &[String], cwd: &str) -> Result<String, AppError> {
    let start = Instant::now();
    log::info!("svn {} (cwd: {})", args.join(" "), cwd);

    let mut cmd = Command::new(svn_path);
    cmd.args(args).current_dir(cwd).stdout(Stdio::piped()).stderr(Stdio::piped());

    let mut child = cmd.spawn().map_err(|e| {
        if e.kind() == std::io::ErrorKind::NotFound {
            AppError::SvnNotFound
        } else {
            AppError::Io(e)
        }
    })?;

    if let Ok(mut guard) = CURRENT_CHILD.lock() {
        *guard = Some(child.try_wait().ok().flatten().map(|_| child).unwrap_or(child));
        drop(guard);
    }

    let output = child.wait_with_output().map_err(AppError::Io)?;
    let elapsed = start.elapsed();
    log::info!("svn 完成 (耗时: {:?}, 退出码: {:?})", elapsed, output.status.code());

    if output.status.success() {
        Ok(String::from_utf8_lossy(&output.stdout).to_string())
    } else {
        let stderr = String::from_utf8_lossy(&output.stderr);
        log::warn!("svn stderr: {}", stderr);
        Err(AppError::SvnCommand(stderr.to_string()))
    }
}

/// 异步执行 SVN 命令（带超时控制 + 取消检测 + 认证检测）
pub async fn run_svn(args: &[&str], cwd: &str) -> Result<String, AppError> {
    if is_cancelled() {
        return Err(AppError::Cancelled);
    }

    let svn_path = get_svn_path();
    let timeout_secs = get_timeout_secs(args);

    let mut all_args: Vec<String> = BASE_SVN_ARGS.iter().map(|s| s.to_string()).collect();
    all_args.extend(args.iter().map(|s| s.to_string()));

    let cwd = cwd.to_string();

    let result = timeout(
        Duration::from_secs(timeout_secs),
        spawn_blocking(move || run_svn_sync(&svn_path, &all_args, &cwd)),
    )
    .await;

    match result {
        Ok(inner) => inner,
        Err(_elapsed) => {
            // 超时后检查 stderr 是否有认证关键词
            let auth_hint = CURRENT_CHILD.lock().ok().and_then(|mut guard| {
                guard.as_mut().and_then(|child| {
                    child.stderr.as_mut().map(|stderr| {
                        let mut reader = BufReader::new(stderr);
                        let mut buf = String::new();
                        let mut auth = false;
                        while reader.read_line(&mut buf).unwrap_or(0) > 0 {
                            if check_auth_hint(&buf) {
                                auth = true;
                            }
                        }
                        auth
                    })
                })
            }).flatten().unwrap_or(false);

            kill_current_process();

            if auth_hint {
                log::warn!("SVN 认证提示超时");
                Err(AppError::SvnAuthFailed(
                    "认证提示超时，请配置 SSH key 或 ~/.subversion/auth/ 凭据缓存".into(),
                ))
            } else {
                log::warn!("SVN 命令超时 ({}s)", timeout_secs);
                Err(AppError::Timeout(format!("SVN 命令执行超时 ({}s)", timeout_secs)))
            }
        }
    }
}

// ── 进程管理 ──────────────────────────────────────────

/// 终止当前 svn 子进程
pub fn kill_current_process() {
    if let Ok(mut guard) = CURRENT_CHILD.lock() {
        if let Some(mut child) = guard.take() {
            log::info!("终止 SVN 进程 pid={:?}", child.id());
            let _ = child.kill();
            let _ = child.wait();
        }
    }
}

/// 终止当前 svn 子进程并执行 cleanup
pub async fn kill_and_cleanup(cwd: &str) {
    kill_current_process();
    log::info!("执行 svn cleanup (cwd: {})", cwd);
    let _ = run_svn(&["cleanup"], cwd).await;
    set_cancelled(false);
}
```

**要点说明：**

1. **`--non-interactive`**：所有命令追加此参数防止 svn 弹出交互式认证对话框
2. **认证检测**（业务设计 §2.2）：超时后检查 stderr 是否含 `Authentication realm` / `Password:` 等关键词，含则返回 `SvnAuthFailed` 而非 `Timeout`
3. **网络检测**：`check_network()` 通过 `TcpStream` 5s 超时尝试连接，`status` 等本地操作不调用
4. **取消流程**：设标志 → kill 子进程 → svn cleanup → 重置标志
5. **路径策略**：debug 模式使用系统 `svn`，release 使用内置路径
