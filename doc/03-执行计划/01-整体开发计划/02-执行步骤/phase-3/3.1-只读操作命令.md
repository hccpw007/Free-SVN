use serde::Deserialize;
use crate::models::error::AppError;
use crate::models::file::FileItem;
use crate::models::repo::RepoInfo;
use crate::svn;

#[derive(Debug, Deserialize)]
pub struct StatusParams {
    pub path: String,
}

#[derive(Debug, Deserialize)]
pub struct InfoParams {
    pub path: String,
}

/// 获取文件变更列表（纯本地，不检测网络）
#[tauri::command]
pub async fn get_status(params: StatusParams) -> Result<Vec<FileItem>, AppError> {

### 3.1B — 新建 `src-tauri/src/commands/diff.rs`

```rust
use serde::Deserialize;
use crate::models::error::AppError;
use crate::svn;
use std::path::Path;

#[derive(Debug, Deserialize)]
pub struct DiffParams {
    pub path: String,
    pub revision1: Option<u64>,
    pub revision2: Option<u64>,
}

#[derive(Debug, serde::Serialize)]
pub struct DiffResult {
    pub content: String,
    pub is_binary: bool,
    pub mime_type: Option<String>,
}

/// 获取文件差异对比文本
#[tauri::command]
pub async fn get_diff(params: DiffParams) -> Result<DiffResult, AppError> {
    svn::executor::validate_path(&params.path)?;

    let path = Path::new(&params.path);
    if is_binary_file(path) {
        return Ok(DiffResult {
            content: String::new(),
            is_binary: true,
            mime_type: None,
        });
    }

    // 构造 svn diff 参数
    let mut args: Vec<String> = vec!["diff".to_string()];
    match (params.revision1, params.revision2) {
        (Some(r1), Some(r2)) => {
            args.push("-r".to_string());
            args.push(format!("{}:{}", r1, r2));
        }
        (Some(r1), None) => {
            args.push("-r".to_string());
            args.push(r1.to_string());
        }
        _ => {}
    }
    args.push(params.path.clone());

    let diff_text = svn::executor::run_svn(
        &args.iter().map(String::as_str).collect::<Vec<&str>>(),
        &params.path,
        None,
    )
    .await?;

    let content = svn::parser::parse_diff(&diff_text)?;
    Ok(DiffResult { content, is_binary: false, mime_type: None })
}

/// 通过扩展名检测二进制文件
fn is_binary_file(path: &Path) -> bool {
    match path.extension().and_then(|e| e.to_str()) {
        Some(ext) => matches!(
            ext.to_lowercase().as_str(),
            "png" | "jpg" | "jpeg" | "gif" | "bmp" | "ico"
                | "pdf" | "doc" | "docx" | "xls" | "xlsx"
                | "ppt" | "pptx"
                | "zip" | "tar" | "gz" | "rar" | "7z"
                | "exe" | "dll" | "so" | "dylib"
                | "jar" | "class"
                | "mp3" | "mp4" | "avi" | "mov" | "wav" | "flac"
                | "ttf" | "otf" | "woff" | "woff2"
        ),
        None => false,
    }
}
```

### 3.1C — 新建 `src-tauri/src/commands/log.rs`

```rust
use serde::Deserialize;
use crate::models::error::AppError;
use crate::svn;
use crate::svn::parser::{BlameLine, LogEntry};

#[derive(Debug, Deserialize)]
pub struct LogParams {
    pub path: String,
    pub limit: Option<u64>,
    pub revision: Option<u64>,
    pub search: Option<String>,
}

#[derive(Debug, Deserialize)]
pub struct BlameParams {
    pub path: String,
    pub revision: Option<u64>,
}

/// 获取版本历史（limit 默认 50，search 仅搜索提交信息）
#[tauri::command]
pub async fn get_log(params: LogParams) -> Result<Vec<LogEntry>, AppError> {
    svn::executor::validate_path(&params.path)?;

    let limit = params.limit.unwrap_or(50);
    let mut args: Vec<String> = vec![
        "log".to_string(),
        "--xml".to_string(),
        "--verbose".to_string(),
        "--limit".to_string(),
        limit.to_string(),
    ];

    if let Some(rev) = params.revision {
        args.push("-r".to_string());
        args.push(format!("{}:1", rev));
    }
    if let Some(ref keyword) = params.search {
        if !keyword.is_empty() {
            args.push("--search".to_string());
            args.push(keyword.clone());
        }
    }
    args.push(params.path.clone());

    let xml = svn::executor::run_svn(
        &args.iter().map(String::as_str).collect::<Vec<&str>>(),
        &params.path,
        None,
    )
    .await?;

    svn::parser::parse_log(&xml)
}

/// 获取文件注解（大文件单独超时 120s）
#[tauri::command]
pub async fn get_blame(params: BlameParams) -> Result<Vec<BlameLine>, AppError> {
    svn::executor::validate_path(&params.path)?;

    let mut args: Vec<String> = vec!["blame".to_string(), "--xml".to_string()];
    if let Some(rev) = params.revision {
        args.push("-r".to_string());
        args.push(rev.to_string());
    }
    args.push(params.path.clone());

    let xml = svn::executor::run_svn(
        &args.iter().map(String::as_str).collect::<Vec<&str>>(),
        &params.path,
        None,
    )
    .await?;

    svn::parser::parse_blame(&xml)
}
```

### 3.1D — 修改 `src-tauri/src/commands/mod.rs`

```rust
pub mod status;
pub mod diff;
pub mod log;
```

**要点说明：**

1. **参数校验**：每个 command 先调用 `validate_path()`
2. **args 构造**：使用 `Vec<String>` 配合 `iter().map(String::as_str)` 转换为 `&[&str]` 传递给 `run_svn`，避免了 `Box::leak` 生命周期问题
3. **二进制检测**：通过文件扩展名判断，前端收到 `is_binary: true` 后显示"该文件为二进制文件，无法显示差异"
4. **limit 默认值**：`params.limit.unwrap_or(50)`，与开发计划一致
5. **`mime_type` 字段对齐说明**：Rust 端 `DiffResult` 的 `mime_type` 类型为 `Option<String>`，TS 端对应 `DiffResult` 接口中 `mimeType?: string`。二者完全对齐（`Option<String>` ↔ `string?`）。当前阶段后端仅通过扩展名检测二进制文件，**不填充 `mime_type` 字段**（始终为 `None`/`undefined`），后续增强时可基于扩展名映射 MIME 类型

## 测试要求

每个 command 函数必须包含 `#[cfg(test)]` 模块进行单元测试：

| 测试对象 | 测试场景 | 说明 |
|---------|---------|------|
| StatusParams 反序列化 | 有效路径 / 空路径 | serde Deserialize 完整性验证 |

| get_diff | 文本差异 / 二进制文件 | 验证 is_binary 检测逻辑 |

**测试方法：** 对于 command 函数，通过构造其参数直接调用，验证返回类型和错误处理。args 构造逻辑可通过单元测试验证 `Result` 类型正确性。不依赖模拟 Tauri 运行时——纯逻辑测试。
