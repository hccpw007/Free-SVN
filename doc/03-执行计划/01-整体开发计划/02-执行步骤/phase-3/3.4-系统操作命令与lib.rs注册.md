### 3.4A — 新建 `src-tauri/src/commands/cancel.rs`

```rust
use crate::models::error::AppError;
use crate::svn;
use tauri::AppHandle;
use tauri::Emitter;

/// 取消当前长操作（业务设计 §2.5）
/// 流程：设取消标志 → kill 子进程 → cleanup → 重置标志 → 推送事件
#[tauri::command]
pub async fn cancel_operation(
    app_handle: AppHandle,
    state: tauri::State<'_, crate::svn::queue::SvnQueue>,
) -> Result<String, AppError> {
    log::info!("取消操作请求");

    // 1. 设置取消标志
    svn::executor::set_cancelled(true);

    // 2. 获取当前工作副本路径（用于 cleanup）
    let cwd = crate::config::store::current_workspace()
        .unwrap_or_default();

    // 3. 终止子进程
    svn::executor::kill_current_process();

    // 4. cleanup 恢复工作副本状态
    if !cwd.is_empty() {
        let _ = svn::executor::run_svn(&["cleanup"], &cwd).await;
    }

    // 5. 重置取消标志
    svn::executor::set_cancelled(false);

    // 6. 释放写操作锁
    state.unlock();

    // 7. 通知前端
    app_handle.emit("operation:completed", serde_json::json!({
        "result": "cancelled",
        "detail": "operation cancelled"
    })).ok();

    log::info!("operation cancelled");
    Ok("operation cancelled".to_string())
}
```

### 3.4C — 新建 `src-tauri/src/commands/network.rs`

包装 `svn::executor::check_network`（原为普通函数，现暴露为 Tauri Command）：

```rust
use crate::models::error::AppError;
use crate::svn;

/// 检测 SVN 服务器网络可达性（业务设计 §10）
/// 通过 TcpStream 5s 超时检测，ping 模式（可被任意 URL 触发）
#[tauri::command]
pub async fn check_network(server_url: Option<String>) -> Result<bool, AppError> {
    // 无参调用时使用默认地址检测网络连通性
    let url = server_url.unwrap_or_else(|| "https://svn.apache.org".to_string());
    match svn::executor::check_network(&url).await {
        Ok(()) => Ok(true),
        Err(AppError::NetworkUnreachable) => Ok(false),
        Err(e) => Err(e),
    }
}
```

### 3.4E ⭐ — 新建 `src-tauri/src/commands/branches.rs`（P0-2 修复）

```rust
use crate::models::error::AppError;
use serde::Deserialize;

/// 列出 SVN 仓库分支列表（svn ls ^/branches）
#[tauri::command]
pub async fn list_branches(url: String) -> Result<Vec<String>, AppError> {
    if url.is_empty() {
        return Ok(vec![]);
    }
    // 从 URL 推断分支根
    let branch_root = if url.contains("/branches/") {
        // 工作副本已位于分支下，定位到 branches/ 父目录
        url.rsplit_once("/branches/").map(|(base, _)| format!("{}/branches", base))
            .unwrap_or_else(|| format!("{}/branches", url.trim_end_matches('/')))
    } else {
        format!("{}/branches", url.trim_end_matches('/'))
    };
    let result = crate::svn::executor::run_svn(
        &["list", &branch_root],
        std::env::temp_dir().to_str().unwrap_or("."),
        None,
    ).await?;
    Ok(result.lines().filter(|l| l.ends_with('/')).map(|l| l.trim_end_matches('/').to_string()).collect())
}
```

**lib.rs 变更说明：**

1. **`.manage(SvnQueue::new())`** — 注册全局单例写操作队列，所有 command 通过 `State<SvnQueue>` 注入
2. **generate_handler! 30 个 command** — 按 3A→3B→3C→3D→3E 批次分组，注释标记便于维护；新增 check_network（network.rs）+ load_settings/save_settings（settings.rs）+ auth 3 个（auth.rs）
3. **所有 plugin 和 setup 逻辑保持不变**

**cancel_operation 要点：**
- 不经过 SvnQueue try_lock（取消操作本身不是写操作）
- 通过 config::store::current_workspace() 获取工作副本路径用于 cleanup
- 始终释放写操作锁（state.unlock()），确保取消后队列可恢复
